10  /**
11   * Indicates that JaCoCo should ignore the annotated type, constructor or
12   * method.
13   *
14   * <p>
15   * <b>Implementation Notice:</b> As of JaCoCo 0.8.3 <i>classes and methods
16   * annotated by annotation whose retention policy is {@code runtime} or
17   * {@code class} and whose simple name contains "Generated" are filtered out
18   * during generation of report</i>.
19   */
20  @Retention(RetentionPolicy.CLASS)
21  @SuppressFBWarnings
22  @Target({ ElementType.CONSTRUCTOR, ElementType.METHOD, ElementType.TYPE })
23  public @interface SuppressJacocoGenerated {
24  	/**
25  	 * Documentation of the reason why JaCoCo is suppressed
26  	 *
27  	 * @return the documentation of the reason why JaCoCo is suppressed
28  	 */
29  	String justification();
30  }
