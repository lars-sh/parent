12  /**
13   * Indicates that all parameters, return values and fields of the class or
14   * package are annotated with {@link NonNull}. The default behavior can be
15   * overwritten using {@link edu.umd.cs.findbugs.annotations.Nullable}.
16   *
17   * <p>
18   * While <a href=
19   * "https://static.javadoc.io/com.google.code.findbugs/jsr305/3.0.2/javax/annotation/ParametersAreNonnullByDefault.html">{@literal @ParametersAreNonnullByDefault}</a>
20   * means "the method parameters in that element are nonnull by default",
21   * {@code @NonNullByDefault} handles parameters and also return values and
22   * fields.
23   *
24   * <p>
25   * {@code @NonNullByDefault} is similar to <a href=
26   * "https://static.javadoc.io/com.github.spotbugs/spotbugs-annotations/3.1.12/edu/umd/cs/findbugs/annotations/DefaultAnnotation.html">{@code @DefaultAnnotation(NonNull.class)}</a>,
27   * but can be used together with Eclipse settings.
28   */
29  @Documented
30  @Retention(RetentionPolicy.CLASS)
31  @Target({ ElementType.PACKAGE, ElementType.TYPE })
32  @DefaultAnnotation(NonNull.class)
33  public @interface NonNullByDefault {
34  	// no annotation type elements needed
35  }
