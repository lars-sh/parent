280 	 * <li>optional fraction

297 	 * </tr>

299 	 * <td>-2m</td>

301 	 * <td>-2 097 152</td>
302 	 * </tr>
303 	 * <tr>
304 	 * <td>0</td>
305 	 * <td></td>
306 	 * <td>0</td>
307 	 * </tr>
308 	 * <tr>
309 	 * <td>5.4k</td>
310 	 * <td>5.4 * 1024</td>
311 	 * <td>5529.6 -&gt; ArithmeticException</td>
312 	 * </tr>
313 	 * <tr>
314 	 * <td>+5_432</td>
315 	 * <td></td>
316 	 * <td>5 432</td>
317 	 * </tr>
318 	 * </table>
319 	 *
320 	 * @param binaryValue binary string
321 	 * @return binary value
322 	 * @throws ParseException on parse failure
323 	 */
324 	@SuppressWarnings("checkstyle:MultipleStringLiterals")
325 	public static BigInteger parseBinaryUnit(final String binaryValue) throws ParseException {
326 		final Optional<Matcher> matcher = Patterns.matches(BINARY_UNIT_PATTERN, binaryValue);
327 		if (!matcher.isPresent()) {
328 			throw new ParseException("Value [%s] does not match binary unit pattern.", binaryValue);
329 		}
330 		final String value = matcher.get().group("value");
331 		if (value == null) {
332 			throw new ParseException("No binary unit value given in string [%s].", binaryValue);
333 		}
334 		final String unit = matcher.get().group("unit");
335 		final BigDecimal multiplicator = unit == null ? BigDecimal.ONE : BINARY_UNITS.get(Strings.toNeutralUpperCase(unit));
336 		return new BigDecimal(replaceAll(value, UNIT_WHITE_SPACE_PATTERN, "")).multiply(multiplicator).toBigIntegerExact();
337 	}
