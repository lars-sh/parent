50  	/**
51  	 * Calls {@code supplier} and forwards its return value. Makes sure any thrown
52  	 * within the supplier is not encapsulated exception inside a
53  	 * {@link InvocationTargetException} or {@link ReflectException}.
54  	 *
55  	 * @param <T>      type of the return value
56  	 * @param supplier functionality to execute
57  	 * @return any value returned by {@code supplier}
58  	 * @throws ReflectException any reflection exception
59  	 * @throws SneakyException  invocation target exceptions
60  	 */
61  	@SuppressWarnings({"checkstyle:IllegalCatch", "PMD.AvoidCatchingGenericException", "PMD.AvoidInstanceofChecksInCatchClause", "PMD.PreserveStackTrace"})
62  	@SuppressFBWarnings(value = {"BC_IMPOSSIBLE_INSTANCEOF", "EXS_EXCEPTION_SOFTENING_NO_CONSTRAINTS"}, justification = "supplier might sneaky throw InvocationTargetException")
63  	private static <T> T extractInvocationTargetExceptions(final Supplier<T> supplier) {
64  		try {
65  			return supplier.get();
66  		} catch (final ReflectException e) {
67  			if (e.getCause() instanceof InvocationTargetException) {
68  				throw new SneakyException(e.getCause().getCause());
69  			}
70  			throw e;
71  		} catch (final Exception e) {
72  			if (e instanceof InvocationTargetException) {
73  				throw new SneakyException(e.getCause());
74  			}
75  			throw e;
76  		}
77  	}
