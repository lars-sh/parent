64  	 * In addition to argument checks {@link Object#equals(Object)} is checked
65  	 * against {@code null}, {@code true} and the object itself while the first two
66  	 * are expected to differ from the object itself.
67  	 *
68  	 * @param constructor function to construct object instances based on given
69  	 *                    arguments
70  	 * @param arguments   declaration of arguments and if changing arguments should
71  	 *                    change equality
72  	 */
73  	@SuppressFBWarnings(value = "PRMC_POSSIBLY_REDUNDANT_METHOD_CALLS", justification = "need to call constructor twice to validate equals")
74  	public static void assertEqualsAndHashCode(final Function<Object[], ?> constructor, final AssertEqualsAndHashCodeArguments arguments) {
75  		final Function<Object[], ?> wrappedConstructor = wrapConstructorExceptions(constructor);
76  		final Object object = wrappedConstructor.apply(arguments.getOriginal().toArray());
77  		final int hashCode = object.hashCode();
78  		assertNotEquals(null, object, "equals null");
79  		assertNotEquals(Boolean.TRUE, object, "equals true");
80  		assertEquals(object, object, "equals this");
81  		final Object sameObject = wrappedConstructor.apply(arguments.getOriginal().toArray());
82  		assertEquals(object, sameObject, "equals original arguments");
83  		assertEquals(hashCode, sameObject.hashCode(), "hashCode original arguments");
84  		IntStream.range(0, arguments.getOriginal().size()).forEach(index -> {
85  			final Consumer3<Object, Object, Supplier<String>> consumer = arguments.isExpectEquality(index) ? org.junit.jupiter.api.Assertions::assertEquals : org.junit.jupiter.api.Assertions::assertNotEquals;
86  			final Object newObject = wrappedConstructor.apply(arguments.getChangedArguments(index).toArray());
87  			// Assert equals
88  			final Supplier<String> equalsMessage = () -> String.format("%sequals changed argument %d", arguments.isExpectEquality(index) ? "" : "not ", index + 1);
89  			consumer.accept(object, newObject, equalsMessage);
90  			// Assert hashCode
91  			final Supplier<String> hashCodeMessage = () -> String.format("hashCode changed argument %d", index + 1);
92  			consumer.accept(hashCode, newObject.hashCode(), hashCodeMessage);
93  		});
94  	}

97  	 * Wraps {@code constructor} to wrap exceptions thrown by that constructor with
98  	 * an {@link AssertionException} showing the used arguments.
99  	 *
100 	 * @param <T>         the constructors class
101 	 * @param constructor the constructor to wrap
102 	 * @return wrapped constructor
103 	 */
104 	@SuppressWarnings({"checkstyle:IllegalCatch", "PMD.AvoidCatchingGenericException"})
105 	@SuppressFBWarnings(value = "EXS_EXCEPTION_SOFTENING_NO_CONSTRAINTS", justification = "throw AssertionException instead")
106 	private static <T> Function<Object[], T> wrapConstructorExceptions(final Function<Object[], T> constructor) {
107 		return arguments -> {
108 			try {
109 				return constructor.apply(arguments);
110 			} catch (final Exception e) {
111 				throw new AssertionException(e, Arrays.stream(arguments).map(Object::toString).collect(joining("], [", "Failed constructing using [", "]")));
112 			}
113 		};
