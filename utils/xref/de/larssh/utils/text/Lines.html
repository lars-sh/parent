163 	 * @param <T>               type of line (most probably {@link String})
164 	 * @param linesIterator     the lines
165 	 * @param isNextConsecutive when returning {@code true} the input line is added
166 	 *                          to an output list, else a new list gets created. The
167 	 *                          first argument is the previous list of lines,
168 	 *                          whereas the second argument is the current line.
169 	 * @return stream of lists containing consecutive lines
170 	 */
171 	public static <T> Stream<List<T>> consecutive(final Iterator<T> linesIterator, final BiPredicate<List<T>, T> isNextConsecutive) {
172 		final PeekableIterator<T> peekableLinesIterator = peekableIterator(linesIterator);
173 		return Iterators.stream(state -> {
174 			final List<T> lines = new ArrayList<>();
175 			while (peekableLinesIterator.hasNext()) {
176 				lines.add(peekableLinesIterator.next());
177 				// Fail fast if the peeked line is not consecutive
178 				if (!isNextConsecutive.test(lines, peekableLinesIterator.peek())) {
179 					return lines;
180 				}
181 			}
182 			return lines.isEmpty() ? state.endOfData() : lines;
183 		});
184 	}

252 	 * @param <V>         type of line
253 	 * @param lines       the lines
254 	 * @param getGroupKey calculates the lines group key
255 	 * @param getLineType calculates the lines type to specify the way groups of
256 	 *                    lines are created and closed. The first argument is the
257 	 *                    previous list of lines, whereas the second argument is the
258 	 *                    current line.
259 	 * @return stream of entries with the group key as key and the grouped lines as
260 	 *         value
261 	 */
262 	@SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
263 	public static <K, V> Stream<Entry<K, List<V>>> grouped(final Iterator<V> lines, final Function<V, K> getGroupKey, final BiFunction<List<V>, V, GroupedLineType> getLineType) {
264 		// Buffer of so-far not-closed groups of lines
265 		final Map<K, List<V>> groups = new LinkedHashMap<>();
266 		return Iterators.stream(state -> {
267 			while (lines.hasNext()) {
268 				final V line = lines.next();
269 				final K groupKey = getGroupKey.apply(line);
270 				List<V> groupOfCurrentLine = groups.computeIfAbsent(groupKey, key -> new ArrayList<>());
271 				final GroupedLineType lineType = getLineType.apply(groupOfCurrentLine, line);
272 				if (lineType == GroupedLineType.END) {
273 					groupOfCurrentLine.add(line);
274 					return Maps.entry(groupKey, groups.remove(groupKey));
275 				}
276 				if (lineType == GroupedLineType.START && !groupOfCurrentLine.isEmpty()) {
277 					groupOfCurrentLine = new ArrayList<>();
278 					groupOfCurrentLine.add(line);
279 					return Maps.entry(groupKey, groups.put(groupKey, groupOfCurrentLine));
280 				}
281 				groupOfCurrentLine.add(line);
282 			}
283 			if (groups.isEmpty()) {
284 				return state.endOfData();
285 			}
286 			final K groupKey = groups.keySet().iterator().next();
287 			return Maps.entry(groupKey, groups.remove(groupKey));
288 		});
289 	}
