37  		}

47  	 * Tries to read a new line from {@code reader}. This method accepts either of
48  	 * {@code \n}, {@code \r\n} or {@code \r} as new line sequence.
49  	 *
50  	 * @param reader a {@link PeekableReader} as data input
51  	 * @return {@code true} if a new line was read., else {@code false}
52  	 * @throws IOException if an I/O error occurs
53  	 */
54  	@PackagePrivate
55  	static boolean readNewLine(final PeekableReader reader) throws IOException {
56  		if (!reader.hasNext()) {
57  			return false;
58  		}
59  		final char character = reader.peek();
60  		if (character != '\r' && character != '\n') {
61  			return false;
62  		}
63  		reader.next();
64  		if (character == '\r' && reader.hasNext() && reader.peek() == '\n') {
65  			reader.next();
66  		}
67  		return true;
68  	}

109 	 * Parses the CSV data given by {@code reader}, starting at the current
110 	 * position.
111 	 *
112 	 * @param reader a {@link PeekableReader} as CSV data input
113 	 * @return an object representing the parsed CSV data
114 	 * @throws IllegalArgumentException on illegal {@code separator} or
115 	 *                                  {@escaper} value
116 	 * @throws IOException              if an I/O error occurs
117 	 */
118 	@SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
119 	private Csv parse(final PeekableReader reader) throws IOException {
120 		assertCsvInput(separator, escaper);
121 		final Csv csv = new Csv();
122 		// Make sure, an empty input results in an empty result with no row
123 		if (!reader.hasNext()) {
124 			return csv.unmodifiable();
125 		}
126 		List<String> currentRow = new CsvRow(csv, csv.size(), new ArrayList<>());
127 		csv.add(currentRow);
128 		while (reader.hasNext()) {
129 			if (readNewLine(reader)) {
130 				// Ignore a trailing new line
131 				if (!reader.hasNext()) {
132 					return csv.unmodifiable();
133 				}
134 				// Add new row
135 				currentRow = new CsvRow(csv, csv.size(), new ArrayList<>());
136 				csv.add(currentRow);
137 			} else {
138 				// Parse the next value and add it to the current row
139 				currentRow.add(parseValue(reader));
140 				// In case of a separator, continue with the next value of the current row
141 				while (reader.hasNext() && reader.peek() == separator) {
142 					reader.next();
143 					currentRow.add(parseValue(reader));
144 				}
145 			}
146 		}
147 		return csv.unmodifiable();
148 	}

150 	/**
151 	 * Parses a single CSV value, starting at the current position and stopping
152 	 * right behind its last character.
153 	 *
154 	 * <p>
155 	 * In case an escaped CSV value comes with leading or trailing spaces, they are
156 	 * stripped off of the actual value.
157 	 *
158 	 * @param reader a {@link PeekableReader} as data input
159 	 * @return the parsed CSV value
160 	 * @throws IOException if an I/O error occurs
161 	 */
162 	@PackagePrivate
163 	@SuppressWarnings("PMD.PrematureDeclaration")
164 	String parseValue(final PeekableReader reader) throws IOException {
165 		final StringBuilder builder = new StringBuilder();
166 		// Check if the given value is escaped. Leading whitespaces need to be read to
167 		// determine that.
168 		final boolean isEscaped = readLeadingWhitespacesAndIsEscaped(reader, builder);
169 		while (reader.hasNext()) {
170 			// Check if control characters, such as the separator or new line characters,
171 			// need to be handled in their special way. If the given value is escaped, the
172 			// escape character needs to precede such characters to take effect.
173 			if (readEscaperAndIsControlCharHandling(reader, isEscaped)) {
174 				// If the given value is escaped and the escaping character is followed by
175 				// whitespaces only, we trim the trailing whitespaces.
176 				final String trailingWhitespaces = isEscaped ? readWhitespaces(reader) : "";
177 				// In case of a separator or new line character the current value was read
178 				// successfully
179 				if (!reader.hasNext() || isSeparatorOrNewLine(reader.peek())) {
180 					return builder.toString();
181 				}
182 				if (isEscaped) {
183 					// Else we simply append the escaping character and further unexpected
184 					// characters.
185 					builder.append(escaper).append(trailingWhitespaces);
186 				}
187 			}
188 			// Append the current character
189 			builder.append(reader.next());
190 		}
191 		return builder.toString();
192 	}

195 	 * Checks if control characters need to be handled.
196 	 *
197 	 * <p>
198 	 * For non-escaped values control characters always need to be handled. For
199 	 * escaped values the value needs to be "closed" using the escaping character
200 	 * prior handling following control characters.
201 	 *
202 	 * @param reader    a {@link PeekableReader} as data input
203 	 * @param isEscaped {@code true} if the current value is escaped, else
204 	 *                  {@code false}
205 	 * @return {@code true} if control characters need to be handled, else
206 	 *         {@code false}
207 	 * @throws IOException if an I/O error occurs
208 	 */
209 	@PackagePrivate
210 	boolean readEscaperAndIsControlCharHandling(final PeekableReader reader, final boolean isEscaped) throws IOException {
211 		if (!isEscaped) {
212 			return true;
213 		}
214 		if (!reader.hasNext() || reader.peek() != escaper) {
215 			return false;
216 		}
217 		reader.next();
218 		return !reader.hasNext() || reader.peek() != escaper;
219 	}

222 	 * Checks if the value, that starts at the reader's position, is escaped.
223 	 * Leading whitespaces need to be read to determine that.
224 	 *
225 	 * <p>
226 	 * The reader is forwarded to the first character of the value. In case of an
227 	 * escaped value, the escaping character is read.
228 	 *
229 	 * <p>
230 	 * Leading whitespaces are trimmed in case of an escaped value, else they are
231 	 * appended to {@code builder}.
232 	 *
233 	 * @param reader  a {@link PeekableReader} as data input
234 	 * @param builder a builder to possibly append whitespaces to.
235 	 * @return {@code true} if the value, that starts at the reader's position, is
236 	 *         escaped, else {@code false}
237 	 * @throws IOException if an I/O error occurs
238 	 */
239 	@PackagePrivate
240 	@SuppressWarnings("PMD.PrematureDeclaration")
241 	boolean readLeadingWhitespacesAndIsEscaped(final PeekableReader reader, final StringBuilder builder) throws IOException {
242 		final String whitespaces = readWhitespaces(reader);
243 		if (reader.hasNext() && reader.peek() == escaper) {
244 			reader.next();
245 			return true;
246 		}
247 		builder.append(whitespaces);
248 		return false;

251 	/**
252 	 * Reads whitespace characters without a special meaning starting at the current
253 	 * position of the reader.
254 	 *
255 	 * <p>
256 	 * Whitespace characters are defined using
257 	 * {@link Characters#isAsciiWhitespace(char)}. Separator, escaping and new line
258 	 * characters are not handled as whitespace character, even if they could be an
259 	 * whitespace character.
260 	 *
261 	 * @param reader a {@link PeekableReader} as data input
262 	 * @return the read whitespace characters
263 	 * @throws IOException if an I/O error occurs
264 	 */
265 	@PackagePrivate
266 	String readWhitespaces(final PeekableReader reader) throws IOException {
267 		final StringBuilder builder = new StringBuilder();
268 		while (reader.hasNext()) {
269 			final char character = reader.peek();
270 			// Stop reading when finding the first non-whitespace character
271 			if (character == escaper || !Characters.isAsciiWhitespace(character) || isSeparatorOrNewLine(character)) {
272 				return builder.toString();
273 			}
274 			builder.append(reader.next());
275 		}
276 		return builder.toString();
277 	}
