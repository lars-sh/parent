22  	 * Pattern describing all single characters retaining the special meaning of a
23  	 * backslash when the backslash is followed by this characters inside a Unix
24  	 * command line arguments.
25  	 */

37  	 * Pattern describing all single space characters inside Windows command line

43  	 * Appends the redirect {@code operation} to {@code file} to {@code builder}.
44  	 * {@code file} is converted into an absolute path and normalized.
45  	 *
46  	 * @param builder   the builder to append to
47  	 * @param operation the redirect operation
48  	 * @param file      the path of the redirection
49  	 */
50  	private static void appendRedirect(final StringBuilder builder, final String operation, final File file) {
51  		final String normalizedPath = file.toPath().toAbsolutePath().normalize().toString();
52  		builder.append(' ').append(operation).append(' ').append(SystemUtils.isWindows() ? escapeArgumentOnWindows(normalizedPath) : escapeArgumentOnUnix(normalizedPath));
53  	}

56  	 * Appends redirect operations of {@code processBuilder} to {@code builder}.
57  	 *
58  	 * @param builder        the builder to append to
59  	 * @param processBuilder the process builder, which information to handle
60  	 */
61  	@SuppressWarnings("PMD.CyclomaticComplexity")
62  	private static void appendRedirects(final StringBuilder builder, final ProcessBuilder processBuilder) {
63  		// Standard Input
64  		final Redirect input = processBuilder.redirectInput();
65  		if (input.type() == Type.READ && input.file() != null) {
66  			appendRedirect(builder, "<", input.file());
67  		}
68  		// Standard Output
69  		final Redirect output = processBuilder.redirectOutput();
70  		if (output.type() == Type.APPEND && output.file() != null) {
71  			appendRedirect(builder, ">>", output.file());
72  		} else if (output.type() == Type.WRITE && output.file() != null) {
73  			appendRedirect(builder, ">", output.file());
74  		}
75  		// Standard Error
76  		if (processBuilder.redirectErrorStream()) {
77  			builder.append(" 2>&1");
78  		} else {
79  			final Redirect error = processBuilder.redirectError();
80  			if (error.type() == Type.APPEND && error.file() != null) {
81  				appendRedirect(builder, "2>>", error.file());
82  			} else if (error.type() == Type.WRITE && error.file() != null) {
83  				appendRedirect(builder, "2>", error.file());
84  			}
85  		}
86  	}

89  	 * Escapes {@code argument} as command line argument for Unix.
90  	 *
91  	 * <p>
92  	 * based on <a href=
93  	 * "https://www.gnu.org/software/bash/manual/bash.html#Double-Quotes">"3.1.2.3
94  	 * Double Quotes" of the Bash Reference Manual</a>
95  	 *
96  	 * @param argument the argument to escape
97  	 * @return the escaped argument
98  	 */
99  	@PackagePrivate
100 	static String escapeArgumentOnUnix(final String argument) {
101 		// No need to quote and escape in case of clearly safe characters
102 		if (Strings.matches(argument, UNIX_SAFE_ARGUMENT_PATTERN)) {
103 			return argument;
104 		}
105 		// Wrap in double quotes and escape characters with special meanings
106 		return '\"' + Strings.replaceAll(argument, UNIX_BACKSLASH_SPECIAL_MEANING_CHARACTER_PATTERN, "\\\\$0") + '\"';
107 	}

109 	/**
110 	 * Escapes {@code argument} as command line argument for Windows.
111 	 *
112 	 * @param argument the argument to escape
113 	 * @return the escaped argument
114 	 */
115 	@PackagePrivate
116 	@SuppressWarnings({"checkstyle:IllegalInstantiation", "checkstyle:XIllegalTypeCustom"})
117 	static String escapeArgumentOnWindows(final String argument) {
118 		final StringBuffer buffer = new StringBuffer();
119 		// The argument needs to be wrapped in double quotes in case of space characters
120 		final boolean quote = argument.isEmpty() || Strings.find(argument, WINDOWS_SPACE_CHARACTER_PATTERN);
121 		if (quote) {
122 			buffer.append('\"');
123 		}
124 		final Matcher matcher = WINDOWS_DOUBLE_QUOTE_PATTERN.matcher(argument);
125 		while (matcher.find()) {
126 			matcher.appendReplacement(buffer, "");
127 			// In case of a double quote trailing backslashes need to be duplicated and the
128 			// found double quote needs to be escaped and appended.
129 			for (int index = buffer.length() - 1; index > -1 && buffer.charAt(index) == '\\'; index -= 1) {
130 				buffer.append('\\');
131 			}
132 			buffer.append("\\\"");
133 		}
134 		matcher.appendTail(buffer);
135 		// In case of quoting trailing backslashes need to be duplicated and the
136 		// trailing double quote needs to be appended.
137 		if (quote) {
138 			for (int index = buffer.length() - 1; index > -1 && buffer.charAt(index) == '\\'; index -= 1) {
139 				buffer.append('\\');
140 			}
141 			buffer.append('\"');
142 		}
143 		return buffer.toString();
144 	}

146 	/**
147 	 * Returns a command to launch {@code processBuilder} via command line.
148 	 *
149 	 * <p>
150 	 * This method is most likely used for logging and user output or debugging.
151 	 *
152 	 * <p>
153 	 * The results of this method depend on the Operating System as command line
154 	 * arguments on Unix and Windows are quoted and escaped differently.
155 	 *
156 	 * @param processBuilder          the process definition
157 	 * @param prependWorkingDirectory if {@code true} the working directory of
158 	 *                                {@code processBuilder} is prepended, else the
159 	 *                                returned value consists of the command line
160 	 *                                only.
161 	 * @return the escaped command
162 	 */
163 	public static String toCommandLine(final ProcessBuilder processBuilder, final boolean prependWorkingDirectory) {
164 		final StringBuilder builder = new StringBuilder();
165 		// Working Directory
166 		if (prependWorkingDirectory) {
167 			builder.append(Nullables.orElseGet(processBuilder.directory(), () -> Paths.get(".").toAbsolutePath().normalize().toString())).append("> ");
168 		}
169 		// Command and Arguments
170 		final UnaryOperator<String> escapeArgument = SystemUtils.isWindows() ? ProcessBuilders::escapeArgumentOnWindows : ProcessBuilders::escapeArgumentOnUnix;
171 		builder.append(processBuilder.command().stream().map(escapeArgument).collect(joining(" ")));
172 		// Redirects (Standard Input, Output, Error)
173 		appendRedirects(builder, processBuilder);
174 		return builder.toString();
175 	}
