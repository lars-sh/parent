10  /**
11   * A {@code MultiReader} concatenates multiple {@link Reader}s into one. That
12   * might be useful for e.g. reading log files of multiple days.
13   *
14   * <p>
15   * The {@code readerIterator} needs to supply the {@link Reader}s to concatenate
16   * one after the other. {@link Reader}s, which have been read, are closed prior
17   * requesting the next.
18   */
19  public class MultiReader extends Reader {
20  	/**
21  	 * Value, that is returned by {@link Reader#read(char[], int, int)} on end of a
22  	 * {@link Reader}.
23  	 */
24  	private static final int END_OF_READER = -1;
25  	/**
26  	 * If {@code true} this {@link MultiReader} reached its end, meaning
27  	 * {@link #readerIterator} has no next element.
28  	 */
29  	private boolean closed = false;
30  	/**
31  	 * The current {@link Reader} or empty if the next needs to be supplied first or
32  	 * if this {@link MultiReader} is closed.
33  	 */
34  	private Optional<Reader> currentReader = Optional.empty();
35  	/**
36  	 * {@link Iterator}, called whenever the next {@link Reader} needs to be
37  	 * supplied. If it has no next element, the {@link MultiReader} is closed.
38  	 */
39  	private final Iterator<Reader> readerIterator;
40  
41  	/**
42  	 * {@inheritDoc}
43  	 */
44  	@Override
45  	public void close() throws IOException {
46  		closed = true;
47  		if (currentReader.isPresent()) {
48  			currentReader.get().close();
49  			currentReader = Optional.empty();
50  		}
51  	}
52  
53  	/**
54  	 * Determines the current {@link Reader} by updating the internal state of this
55  	 * {@link MultiReader} if necessary.
56  	 *
57  	 * @return the current {@link Reader} or empty if closed
58  	 */
59  	@SuppressWarnings({"checkstyle:SuppressWarnings", "resource"})
60  	private Optional<Reader> getCurrentReader() {
61  		if (closed) {
62  			return Optional.empty();
63  		}
64  		if (currentReader.isPresent()) {
65  			return currentReader;
66  		}
67  		currentReader = readerIterator.hasNext() ? Optional.of(readerIterator.next()) : Optional.empty();
68  		if (!currentReader.isPresent()) {
69  			closed = true;
70  		}
71  		return currentReader;
72  	}

74  	/**
75  	 * {@inheritDoc}
76  	 */
77  	@Override
78  	@SuppressWarnings("checkstyle:SuppressWarnings")
79  	public int read(@Nullable final char[] outputBuffer, final int offset, final int length) throws IOException {
80  		final Optional<Reader> reader = getCurrentReader();
81  		if (!reader.isPresent()) {
82  			return END_OF_READER;
83  		}
84  		// Read from current reader
85  		@SuppressWarnings("resource")
86  		final int chars = reader.get().read(outputBuffer, offset, length);
87  		if (chars == length) {
88  			return chars;
89  		}
90  		// If less characters than expected have been read from the current reader,
91  		// close it and continue with the next
92  		currentReader = Optional.empty();
93  		reader.get().close();
94  		// Read additional chunk of characters
95  		final int nonNegativeChars = chars == END_OF_READER ? 0 : chars;
96  		final int nextChars = read(outputBuffer, offset + nonNegativeChars, length - nonNegativeChars);
97  		return nonNegativeChars + (nextChars == END_OF_READER ? 0 : nextChars);
98  	}
