105 	 * Sorts a {@link Map} based on a given {@link Entry} {@link Comparator}.
106 	 *
107 	 * <p>
108 	 * Currently returned map is implemented as {@link java.util.LinkedHashMap}
109 	 * though that might change.
110 	 *
111 	 * @param <K>        the type of keys maintained by this map
112 	 * @param <V>        the type of mapped values
113 	 * @param unsorted   probably unsorted map
114 	 * @param comparator entry based comparator
115 	 * @return sorted map based on the given comparator
116 	 */
117 	public static <K, V> Map<K, V> sort(final Map<K, V> unsorted, final Comparator<Entry<K, V>> comparator) {
118 		return unsorted.entrySet().stream().sorted(comparator).collect(toLinkedHashMap(Entry::getKey, Entry::getValue));
119 	}
